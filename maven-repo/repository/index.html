🔧 使用方法

1. 添加仓库

repositories {
    maven { url = uri("https://yourname.github.io/nativescript-android-runtime/maven-repo/repository") }
}

2. 添加依赖

dependencies {
    implementation("com.nativescript:android-runtime:1.0.0")
    implementation("com.nativescript:widgets:1.0.0")
}
